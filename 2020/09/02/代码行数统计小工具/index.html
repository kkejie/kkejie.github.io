import os
from tkinter import *


def get_count_code_line(dir_path):       # 定义统计代码行的函数
    file_count = 0      # py文件个数
    all_line_count = 0      # 代码行数
    empty_line_count = 0        # 空行数
    comment_line_count = 0          # 注释行数
    for root, dirs, files in os.walk(dir_path):
        for file_name in files:
            file_path = os.path.join(root, file_name)
            # 从路径最后三个字符判断该文件是否为python文件
            if file_path[-3:] == ".py":
                file_count += 1
                with open(file_path, 'r', encoding="utf-8") as fp:
                    # 从文件中逐行提取
                    for line in fp:
                        all_line_count += 1
                        # 判断是否为空行
                        if line.strip() == "":
                            empty_line_count += 1
                        # line.lstrip()  去除每行左边的空白
                        # 判断是否为注释
                        elif line.lstrip()[0] == "#":
                            comment_line_count += 1

    return file_count, all_line_count, empty_line_count, comment_line_count


windows = Tk()
windows.title("代码统计小工具")
windows.geometry("550x250")
L1 = Label(windows, text="输入目录路径，点【提交】后进行代码统计")
E1 = Entry(windows, bd=5, width=200)
L1.pack()
E1.pack()


def clicked():
    dir_path = E1.get()
    msg_code_lines = get_count_code_line(dir_path)
    file_count = msg_code_lines[0]
    all_line_count = msg_code_lines[1]
    empty_line_count = msg_code_lines[2]
    comment_line_count = msg_code_lines[3]
    L2.configure(text="python文件总个数：%s 个\n总代码行数：%s 行\n空行数：%s 行\n注释行数： %s 行" % (
        file_count, all_line_count, empty_line_count, comment_line_count))


button = Button(windows, text="提交", command=clicked)
button.pack()
L2 = Label(windows, text="", bg='white', width=200, height=10)
L2.pack()
windows.mainloop()
